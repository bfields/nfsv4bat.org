    Example #2 - Using RandomAccessFile Interface
    import java.io.*;
    import java.net.*;
    import sunsoft.onc.WebNFS.*;
    //
    // create connection to host
    //
    URL url = new URL( "hostname","/tmp/info.txt", "nfs");
    //
    // create random access file
    //
    RandomAcessFile nraf = new RandomAcessFile(url,"r");
    //
    // use raf
    //
    int count = 0;
    int val = 0;
    count=10;
    nraf.seek(count);
    System.out.println("Val at position " + count + " is " + (byte)nraf.read());
    System.out.println("Current Position is " + nraf.getFilePointer());
    nraf.seek(0);
    //
    // read through file
    //
    for ( count = 0 ; count < 100 ; count ++ ) {
    val = nraf.read();
    if ( val == -1 ) break;
    System.out.write(val);
    }
