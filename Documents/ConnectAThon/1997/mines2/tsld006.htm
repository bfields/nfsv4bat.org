    import java.io.*;
    import java.net.*;
    import sunsoft.onc.WebNFS.*;
    NFSFileAccessorFactory f = null;
    //
    // create nfs protocol factory
    //
    f = new NFSFileAccessorFactory();
    //
    // register nfs protocol
    //
    java.io.File.addFileAccessorFactory("nfs",f);
    //
    // create File using NFS protocol
    //
    File nfile = new File(new URL("nfs://jumble/file.txt"));
    //
    // test to see if file is directory
    //
    if (nfile.isDir()) System.out.println("File is directory");
    //
    // get file system / protocol specific interface from factory
    //
    NFSFileAccessor nfa = nfile.getFileAccessor();
    //
    // use interface to get file system /protocol specific information
    //
    System.out.println("File GID is " + nfa.getGID());
